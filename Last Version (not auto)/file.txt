//get list of interface names from output of netsh command
for /f "tokens=3* delims= " %a in ('netsh Interface show interface ^|find /I "Connected"') do echo %b

//for batch file with the same reason before
for /f "tokens=3* delims= " %%a in ('netsh Interface show interface ^|find /I "Connected"') do echo %%b


//getting it into a variable
for /f "tokens=3* delims= " %%a in ('netsh Interface show interface ^|find /i "Connected" ^|find /I "LAN"  ') do set Interface=%%b
